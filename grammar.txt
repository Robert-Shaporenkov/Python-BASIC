                :  comp_expr ((KEYWORD:and|KEYWORD:or) comp_expr)*

                :  arith_expr ((EE_LT_GT_LTE_GTE) arith_expr)*

factor          :  (PLUS|MINUS) factor
                :  power

                :  LPAREN expr RPAREN
